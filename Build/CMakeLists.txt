cmake_minimum_required(VERSION 3.27.0)
set(CMAKE_CXX_STANDARD 20)
project(iris)
file(GLOB_RECURSE lsp ../LSP/*.cpp)
add_executable(iris ../Iris.cpp ${lsp})
if(MSVC)
    target_compile_definitions(iris PRIVATE iris_windows=)
endif()
target_compile_options(iris PRIVATE -flto -O3 -fno-rtti -fno-exceptions
-fomit-frame-pointer)
target_link_options(iris PRIVATE -flto)
